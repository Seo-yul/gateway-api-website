{{- $file := .Get "file" -}} {{- if not $file -}} {{- errorf "include shortcode requires a 'file' parameter" -}} {{- end -}} {{- $ext := path.Ext $file -}} {{- $lang := "text" -}} {{- if or (eq $ext ".yaml") (eq $ext ".yml") -}} {{- $lang = "yaml" -}} {{- else if eq $ext ".go" -}} {{- $lang = "go" -}} {{- else if eq $ext ".json" -}} {{- $lang = "json" -}} {{- else if eq $ext ".sh" -}} {{- $lang = "bash" -}} {{- else if eq $ext ".py" -}} {{- $lang = "python" -}} {{- else if eq $ext ".js" -}} {{- $lang = "javascript" -}} {{- end -}} {{- $content := "" -}} {{- if fileExists $file -}} {{- $raw := readFile $file -}} {{- $lines := split $raw "\n" -}} {{- $filtered := slice -}} {{- range $lines -}} {{- if not (hasPrefix . "#$") -}} {{- $filtered = $filtered | append . -}} {{- end -}} {{- end -}} {{- $content = delimit $filtered "\n" -}} {{- end -}} {{- if $content -}}
{{ $file }}
{{- highlight $content $lang "" -}}
{{- else -}}
File not found: {{ $file }}
{{- end -}}
